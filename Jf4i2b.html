문제

Autoboxing과 Unboxing의 예제 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요 
num1 = 20, num2 = 20


num1 = 20, num2 = 20


num1 = 20, num2 = 20

코드 빈칸 채우기

public class AutoboxingUnboxingExample  
{ 
	public static void main( String[] args )  
	{ 
		 num1 = 10; 
 
		 num2 = num1; 
		num2 = .( "20" ); 
 
		num1 = num2; 
		System.out.println( "num1 = " + num1 + ", num2 = " + num2); 
	} 
}
