문제

클래스 객체 참조형 
다양한 연산자를 실습하고 그 결과를 출력하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
  color = green
( color == green ) = false,     ( color == blue ) = false,      ( color == red ) = false
  color.equals( green ) = true,   color.equals( blue ) = false,   color.equals( red ) = false
  color.compareTo( green ) = 0,   color.compareTo( blue ) = 5,    color.compareTo( red ) = -11

프로그램 코드

				
	public class ReferenceOperators
	{
1		public static void main( String[] args ) {
			String color = new String( "green" );
			String green = new String( "green" );
			String blue = new String( "blue" );
2			String red = new String( "red" );
3			System.out.println( "  color = green" );
			
4			System.out.print( "( color == green ) = " + ( color == green ) + ",     " );
5			System.out.print( "( color == blue ) = " + ( color == blue ) + ",      "  );
6			System.out.print( "( color == red ) = " + ( color == red ) + "\n" );
			
7			System.out.print( "  color.equals( green ) = " + color.equals( green ) + ", "  );
8			System.out.print( "  color.equals( blue ) = " + color.equals( blue ) + ", "  );
9			System.out.print( "  color.equals( red ) = " + color.equals( red )  + "\n" );
			
10			System.out.print( "  color.compareTo( green ) = " + color.compareTo( green )  + ", " );
11			System.out.print( "  color.compareTo( blue ) = " + color.compareTo( blue ) + ",  "  );
12			System.out.print( "  color.compareTo( red ) = " + color.compareTo( red ) + "\n" );
13		}
	}
