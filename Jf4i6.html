프로그램시작 
   에어컨 리모컨 객체 생성 
   에어컨 리모컨의 입력에 따라 에어컨을 제어하고 그 결과를 출력 
      전원을 입력하면 전원을 켜거나 끄기 
      상을 입력하면 온도를 +1 이동 
      하를 입력하면 온도를 -1 이동 
      좌를 입력하면 풍량을 -1 이동 
      우를 입력하면 풍량을 +1 이동 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Common/remoteControl/RemoteControl.java 

	package remoteControl; 

	 

	/** 

	 * 전원, 상, 하, 좌, 우 버튼이 있는 리모컨  

	 * @author So Young Park 

	 * @version 0.8 

	 */	 

	public interface RemoteControl 

	{ 

		/** 

		 * 전원 버튼 켜짐(ON) 상태 관련 상수 

		 */ 

		public final boolean ON = true; 

		/** 

		 * 전원 버튼 꺼짐(OFF) 상태 관련 상수 

		 */ 

		public final boolean OFF = false;	 

		/** 

		 * 리모컨에서 전원버튼 눌렀을 때 작동 

		 * @return 리모컨 조정 후 가전제품 상태   

		 */ 

		public String clickPower(); 

		/** 

		 * 리모컨에서 상(△) 버튼을 눌렀을 때 작동 

		 * @return 리모컨 조정 후 가전제품 상태   

		 */ 

		public String clickUp(); 

		/** 

		 * 리모컨에서 하(▽) 버튼을 눌렀을 때 작동 

		 * @return 리모컨 조정 후 가전제품 상태   

		 */ 

		public String clickDown(); 

		/** 

		 * 리모컨에서 좌(◁) 버튼을 눌렀을 때 작동 

		 * @return 리모컨 조정 후 가전제품 상태   

		 */ 

		public String clickLeft(); 

		/** 

		 * 리모컨에서 우(▷) 버튼을 눌렀을 때 작동 

		 * @return 리모컨 조정 후 가전제품 상태   

		 */ 

		public String clickRight(); 

	}

	package remoteControl; 

	 

	/** 

	 * AirConditioner 클래스는 에어컨 리모컨을 모델링  

	 * @author 박소영 

	 * 

	 */ 

	public class AirConditioner implements RemoteControl  

	 

	{				 

		// 에어컨 전원, 온도, 풍량 변수  

		private boolean power;						 

		private int temperature;						 

		private int windStrength;							 

		 

		/** 

		 * AirConditioner 생성자로 에어컨 상태 초기화 

		 * @return 생성자이므로 해당사항 없음 

		 */ 

1		public AirConditioner()									 

		{ 

			this.power = OFF;						 

			this.temperature = 18;						 

2			this.windStrength = 10;						 

		} 

		 

		/** 

		 * 에어컨 리모컨에서 전원버튼 눌렀을 때 전원을 켜거나 끄기 

		 * @return 전원 상태    

		 */ 

3		public String clickPower()					 

		{ 

			if( this.power == OFF )					 

			{ 

				this.power = ON;					 

				return "전원을 켭니다"; 

			} 

			else  

			{ 

				this.power = OFF;					 

				return "전원을 끕니다"; 

			} 

4		} 

		 

		/** 

		 * 에어컨 리모컨에서 상(△) 버튼을 눌렀을 때 온도를 +1 이동	 

		 * @return 온도 상태  

		 */ 

5		public String clickUp()						 

		{ 

6			return "현재 온도는 " + ( ++this.temperature ) + "도입니다";		 

		} 

		 

		/** 

		 * 에어컨 리모컨에서 하(▽) 버튼을 눌렀을 때 온도를 -1 이동 

		 * @return 온도 상태  

		 */ 

7		public String clickDown()					 

		{ 

8			return "현재 온도는 " + ( --this.temperature ) + "도입니다";		 

		} 

		 

		/** 

		 * 에어컨 리모컨에서 좌(◁) 버튼을 눌렀을 때 풍량을 -1 이동	 

		 * @return 풍량 상태  

		 */ 

9		public String clickLeft()					 

		{ 

10			return "현재 풍량은 " + ( --this.windStrength ) + "입니다";		 

		} 

		 

		/** 

		 * 에어컨 리모컨에서 우(▷) 버튼을 눌렀을 때 풍량을 +1 이동 

		 * @return 풍량 상태  

		 */ 

11		public String clickRight()					 

		{ 

12			return "현재 풍량은 " + ( ++this.windStrength ) + "입니다";		 

		} 

	}
