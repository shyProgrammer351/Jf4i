public  TV  RemoteControl  
{				 
	private boolean power;	  
	private int channel; 		  
	private int volume;			  
	 
	public TV()					  
	{ 
		.power = OFF;			  
		.channel = 100; 	 
		.volume = 10;		 
	} 
	 
	public String clickPower()	  
	{ 
		if( .power == OFF )	 			  
		{ 
			.power = ON;		 				  
			return "TV 전원을 켭니다"; 
		} 
		else  
		{ 
			.power = OFF;		 			 	 
			return "TV 전원을 끕니다"; 
		} 
	} 
	 
	public String clickUp()		  
	{ 
		return "현재 채널은 " + ( ++.channel ) + "번입니다";		 				  
	} 
	 
	public String clickDown()				  
	{ 
		return "현재 채널은 " + ( --.channel ) + "번입니다";		 			  
	} 
	 
	public String clickLeft()			  
	{ 
		return "현재 음량은 " + ( --.volume ) + "입니다";		 			  
	} 
	 
	public String clickRight()		  
	{ 
		return "현재 음량은 " + ( ++.volume ) + "입니다";		 			  
	} 
}

public  RemoteControl_All_Main  
{ 
	public static void main (String[] args)  
	{ 
		Scanner scan =  Scanner( System.in ); 
		String result = ""; 
 
		RemoteControl rc[] = { 
				 TV(),						 								 
				 AirConditioner(),								 
				 RoboticVacuumCleaner() };											 
		 
			do 
			{ 
				System.out.print( "리모컨의 상, 하, 좌, 우, 전원, 종료 중 하나를 입력하세요 : "); 
				result = scan.next(); 
 
				if ( result.equals( "전원" ) ) 
				{ 
					for ( int i = 0; i < rc.length; i++ ) 
					{ 
						System.out.println( 
							rc[i].clickPower() );	  
					} 
				} 
				else if ( result.equals( "상" ) ) 
				{ 
					for ( int i = 0; i < rc.length; i++ ) 
					{ 
						System.out.println( 
							rc[i].clickUp() );	  
					} 
				} 
				else if ( result.equals( "하" ) ) 
				{ 
					for ( int i = 0; i < rc.length; i++ ) 
					{ 
						System.out.println( 
							rc[i].clickDown() );	  
					} 
				} 
				else if ( result.equals( "좌" ) ) 
				{ 
					for ( int i = 0; i < rc.length; i++ ) 
					{ 
						System.out.println( 
							rc[i].clickLeft() );	  
					} 
				} 
				else if ( result.equals( "우" ) ) 
				{ 
					for ( int i = 0; i < rc.length; i++ ) 
					{ 
						System.out.println( 
							rc[i].clickRight() );	  
					} 
				}				 
 
			} while( !result.equals( "종료" ) ); 
 
		return;		 
	} 
}
