 remoteControl; 
 
  RemoteControl 
{ 
	  boolean ON = true; 
	  boolean OFF = false;	 
	 String clickPower(); 
	 String clickUp(); 
	 String clickDown(); 
	 String clickLeft(); 
	 String clickRight(); 
} 
 remoteControl; 
 
  AirConditioner  RemoteControl  
 
{				 
	private boolean power;						 
	private int temperature;						 
	private int windStrength;							 
	 
	 AirConditioner()									 
	{ 
		.power = OFF;						 
		.temperature = 18;						 
		.windStrength = 10;						 
	} 
	 
	 String clickPower()					 
	{ 
		if( .power == OFF )					 
		{ 
			.power = ON;					 
			return "전원을 켭니다"; 
		} 
		else  
		{ 
			.power = OFF;					 
			return "전원을 끕니다"; 
		} 
	} 
	 
	 String clickUp()						 
	{ 
		return "현재 온도는 " + ( ++.temperature ) + "도입니다";		 
	} 
	 
	 String clickDown()					 
	{ 
		return "현재 온도는 " + ( --.temperature ) + "도입니다";		 
	} 
	 
	 String clickLeft()					 
	{ 
		return "현재 풍량은 " + ( --.windStrength ) + "입니다";		 
	} 
	 
	 String clickRight()					 
	{ 
		return "현재 풍량은 " + ( ++.windStrength ) + "입니다";		 
	} 
} 
 java.util.Scanner; 
 remoteControl.AirConditioner; 
  AirConditioner_Main 								 
{ 
	 static void main( String[] args ) 	 
	{ 
		Scanner scan =  Scanner( System.in ); 
		String result = ""; 
		 
		AirConditioner aircon =  
		                         AirConditioner();

			result = scan.next(); 
			 
			if ( result.equals( "전원" ) ) 
			{ 
				System.out.println(

			}

			}
