문제

A양이 가위바위보를 선택하면 B군이 임의로 가위바위보를 선택하고 지는 가위바위보의 승패를 출력하는 문제입니다 이를 해결하는 다음 프로그램을 해석하세요 
가위, 바위, 보 중 하나를 입력하세요 : 가위
플레이어 A는 가위를 냈습니다.
플레이어 B는 보를 냈습니다.
판정결과는 플레이어 B가 이겼습니다


가위, 바위, 보 중 하나를 입력하세요 : 바위
플레이어 A는 바위를 냈습니다.
플레이어 B는 보를 냈습니다.
판정결과는 플레이어 A가 이겼습니다


가위, 바위, 보 중 하나를 입력하세요 : 보
플레이어 A는 보를 냈습니다.
플레이어 B는 가위를 냈습니다.
판정결과는 플레이어 A가 이겼습니다

	{

		public final int ROCK = 0;  

		public final int PAPER = 1;		  

		public final int SCISSORS = 2;	  

	 

		// 플레이어 A, 플레이어 B는 가위바위보 중 하나를 값으로 선택 

		private String playerA;		  

		private String playerB;  

	 

		/** 

		 * 가위바위보 게임을 초기화하는 생성자 

		 */ 

1		public RockPaperScissors()	  

		{	 

			this.playerA = "가위";	  

2			this.playerB = "바위";	  

		} 

		 

		/** 

		 * 플레이어 A 값은 키보드에서 입력받고 플레이어 B 값은 임의로 입력받아 가위바위보 게임의 승패를 판정 

		 * @param scan : 키보드 입력을 위한 Scanner 변수 전달 

		 * @return 가위바위보 게임을 진행한 결과 

		 */ 

		// 가위바위보 게임 진행하는 메서드 

3		public String play( Scanner scan )	  

		{	 

			// 플레이어 A는 키보드에서 입력받아 가위바위보 선택  

4			this.playerA = 		  

5			               getRockPaperScissors( scan );  

			 

			// 플레이어 B의 임의로 가위바위보 선택 

6			this.playerB = 				  

7			               getRockPaperScissors();

8			String result =  

9			               getResult( this.playerA, this.playerB );		  

	 

10			return result;

		 * 플레이어 A 값은 전달받고 플레이어 B 값은 임의로 입력받아 가위바위보 게임의 승패를 판정 

		 * @param playerA : 플레이어 A의 가위바위보 중 하나 

		 * @return 가위바위보 게임을 진행한 결과 

		 */ 

11		public String play( String playerA )	  

		{ 

			// 플레이어 A는 전달받아 가위바위보 선택 

12			this.playerA = playerA;		  

			 

			// 플레이어 B의 임의로 가위바위보 선택  

13			this.playerB = 				  

14			               getRockPaperScissors();	 

	 

			// 가위바위보 게임을 진행한 결과를 반환 

15			String result =  

16			               getResult( this.playerA, this.playerB );			  

	 

17			return result; 

		} 

		 

		/** 

		 * 가위바위보 게임을 진행한 결과 

		 * @param playerA : 플레이어 A의 가위바위보 중 하나 

		 * @return 가위바위보 게임을 진행한 결과 

		 */ 

18		private String getResult( String playerA, String playerB )	  

		{ 

			// 플레이어 A와 플레이어 B의 현재 상태  

			String result = ""; 

			result += "플레이어 A는 " + playerA +"를 냈습니다.\n"; 

19			result += "플레이어 B는 " + playerB +"를 냈습니다.\n"; 

			 

			// 플레이어 A와 플레이어 B의 가위바위보 승패 판정  

20			result += "판정결과는 "  

21			                  + judge( playerA, playerB ) + "\n";   

			 

22			return result; 

		} 

	 

		/** 

		 * 가위바위보 게임을 진행한 결과 

		 * @return 가위바위보 게임을 진행한 결과 

		 */ 

23		private String getResult()  

		{ 

24			return 

25			       getResult( this.playerA, this.playerB );		  

		} 

	 

		/** 

		 * 키보드에서 입력받아 가위바위보 선택 

		 * @param scan : 키보드 입력을 위한 Scanner 변수 전달 

		 * @return 가위바위보 중 하나 

		 */ 

26		private String getRockPaperScissors( Scanner scan )  

		{ 

			String result = ""; 

			 

			do 

			{ 

				System.out.print( "가위, 바위, 보 중 하나를 입력하세요 : "); 

				// 가위를 입력한다고 가정  

				result = scan.next(); 

				 

			} while( !result.equals( "가위" ) && !result.equals( "바위" ) && !result.equals( "보" ) ); 

				 

27			return result; 

		} 

	 

		/** 

		 * 임의로 가위바위보 선택 

		 * @return 가위바위보 중 하나 

		 */ 

28		private String getRockPaperScissors()			  

		{ 

			// 가위바위보를 0~2로 표현하고, 임의로 1(보)을 선택했다고 가정  

			int num = (int)(Math.random()*3); 

			String result = ""; 

			 

			if ( num == SCISSORS ) 

			{ 

				result = "가위"; 

			} 

			else if ( num == ROCK ) 

			{ 

				result = "바위"; 

			} 

			else 

			{ 

				result = "보";  

			} 

			 

29			return result;  

		} 

				 

		/** 

		 * 플레이어 A와 플레이어 B의 가위바위보 승패 판정 

		 * @param playerA : 플레이어 A의 가위바위보 선택 결과 

		 * @param playerB : 플레이어 A의 가위바위보 선택 결과 

		 * @return 가위바위보 승패 판정 결과 

		 */ 

30		protected String judge( String playerA, String playerB )	  

		{ 

			String result = ""; 

			if ( playerA.equals( playerB ) ) 

			{ 

				result = "플레이어 A와 플레이어 B가 비겼습니다"; 

			} 

			else if ( ( playerA.equals( "가위" ) && playerB.equals( "보" ) )  

					|| ( playerA.equals( "바위" ) && playerB.equals( "가위" ) ) 

					|| ( playerA.equals( "보" ) && playerB.equals( "바위" ) ) ) 

			{ 

				result = "플레이어 A가 이겼습니다"; 

			} 

			else 

			{ 

				result = "플레이어 B가 이겼습니다"; 

			}	 

			 

31			return result; 

		} 

	}
