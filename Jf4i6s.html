	package remoteControl;


	public interface RemoteControl

	{

		public final boolean ON = true;

		public final boolean OFF = false;	

		public String clickPower();

		public String clickUp();

		public String clickDown();

		public String clickLeft();

		public String clickRight();

	}


	package remoteControl;


	public class AirConditioner implements RemoteControl 


	{				

		private boolean power;						

		private int temperature;						

		private int windStrength;							

		

1		public AirConditioner()									

		{

			this.power = OFF;						

			this.temperature = 18;						

2			this.windStrength = 10;						

		}

		

3		public String clickPower()					

		{

			if( this.power == OFF )					

			{

				this.power = ON;					

				return "전원을 켭니다";

			}

			else 

			{

				this.power = OFF;					

				return "전원을 끕니다";

			}

4		}

		

5		public String clickUp()						

		{

6			return "현재 온도는 " + ( ++this.temperature ) + "도입니다";		

		}

		

7		public String clickDown()					

		{

8			return "현재 온도는 " + ( --this.temperature ) + "도입니다";		

		}

		

9		public String clickLeft()					

		{

10			return "현재 풍량은 " + ( --this.windStrength ) + "입니다";		

		}

		

11		public String clickRight()					

		{

12			return "현재 풍량은 " + ( ++this.windStrength ) + "입니다";		

		}

	}
