package remoteControl; 
 
public  RemoteControl 
{ 
	public final boolean ON = true; 
	public final boolean OFF = false;	 
	public String clickPower(); 
	public String clickUp(); 
	public String clickDown(); 
	public String clickLeft(); 
	public String clickRight(); 
} 
package remoteControl; 
import java.util.Scanner; 
 
public  TV  RemoteControl  
{				 
	private boolean power;	  
	private int channel;  
	private int volume;		  
	 
	public TV()					  
	{ 
		.power = OFF;		  
		.channel = 100;  
		.volume = 10;		  
	} 
	 
	public String clickPower()	  
	{ 
		if( .power == OFF )	  
		{ 
			.power = ON;		  
			return "전원을 켭니다"; 
		} 
		else  
		{ 
			.power = OFF;		  
			return "전원을 끕니다"; 
		} 
	} 
	 
	public String clickUp()		  
	{ 
		return "현재 채널은 " + ( ++.channel ) + "번입니다";		  
	} 
	 
	public String clickDown()				  
	{ 
		return "현재 채널은 " + ( --.channel ) + "번입니다";		  
	} 
	 
	public String clickLeft()			  
	{ 
		return "현재 음량은 " + ( --.volume ) + "입니다";		  
	} 
	 
	public String clickRight()		  
	{ 
		return "현재 음량은 " + ( ++.volume ) + "입니다";		  
	} 
} 
import java.util.Scanner; 
import remoteControl.TV; 
public  TV_Main 							  
{ 
	public static void main( String[] args ) 	  
	{ 
		Scanner scan =  Scanner( System.in ); 
		String result = ""; 
		 
		TV tv =  
		         TV();						  
 
		do 
		{ 
			System.out.print( "TV 리모컨의 상, 하, 좌, 우, 전원, 종료 중 하나를 입력하세요 : "); 
			result = scan.next(); 
			 
			if ( result.equals( "전원" ) ) 
			{ 
				System.out.println( 
				                    tv.clickPower() ); 
			} 
			else if ( result.equals( "상" ) ) 
			{ 
				System.out.println( 
				                    tv.clickUp() ); 
			} 
			else if ( result.equals( "하" ) ) 
			{ 
				System.out.println( 
				                    tv.clickDown() ); 
			} 
			else if ( result.equals( "좌" ) ) 
			{ 
				System.out.println( 
				                    tv.clickLeft() ); 
			} 
			else if ( result.equals( "우" ) ) 
			{ 
				System.out.println( 
				                    tv.clickRight() ); 
			}				 
 
		} while( !result.equals( "종료" ) ); 
						 
		return;		 
	} 
}
